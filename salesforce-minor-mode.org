(defun sfmm:binding--key (key)
  (kbd (concat sfmm:keymap-prefix " " key)))

(defun sfmm:binding--key-org (key)

*** HOOK FUNCTIONS
#+begin_src emacs-lisp :tangle yes
(defun sfmm:hook-on-handler ()
  (setq org-default-alias (sfdx-org-default))
  (when org-default-alias
    (add-to-list 'mode-line-misc-info '(" " org-default-alias " "))
    (sfdx-org-connect-status)))


(defun sfmm:hook-off-handler ()
  (setq org-default-alias (sfdx-org-default))
  (when org-default-alias
    (setq mode-line-misc-info
        (delete '(" " org-default-alias " ") mode-line-misc-info))))

#+end_src

** MINOR MODE
#+begin_src emacs-lisp :tangle yes
(define-minor-mode salesforce-minor-mode
  "Toggles global salesforce minor mode."
  nil ; Inital value, nil for disabled
  :global t
  :group 'salesforce
  :lighter " salesforce-minor-mode"
  :keymap
  (list (cons (sfmm:binding--key "r") #'sfdx-retrieve-file)
        (cons (sfmm:binding--key "d") #'sfdx-push-file)
        (cons (sfmm:binding--key-auth "a") #'sfdx-authorize-web)
        (cons (sfmm:binding--key-apex "e") #'sfdx-execute-apex-code)
        (cons (sfmm:binding--key-query "s") #'sfdx-fetch-record-through-soql)
        (cons (sfmm:binding--key-org "a") #'sfdx-display-all-scratch-orgs)
        (cons (sfmm:binding--key-org "A") #'sfdx-display-all-non-scratch-orgs)
        (cons (sfmm:binding--key-org "o") #'sfdx-org-open))
        


  (add-hook 'salesforce-minor-mode-on-hook #'sfmm:hook-on-handler)

  (add-hook 'salesforce-minor-mode-off-hook #'sfmm:hook-off-handler))

(provide 'salesforce-minor-mode)

#+end_src
