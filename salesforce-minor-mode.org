   "Get errors from json data"
   (let ((component-failures (sfdx--helper:get-data-hashtable "result.deployedSource" data)))

      (lambda (component)
        (let ((problem (gethash "error" component))

              (file-name (gethash "filePath" component)))

                  file-name

(defun sfmm:display-all-org (org-type)
  "Display all current connect org"
  (let ((command (sfdx--helper:generate-command (list sfmm:org-command-alias "list" "--json"))))

    (sfdx--helper:make-sync-process
     :command command
     :handle-success-lambda
     (lambda (process json-instance buffer)
       (let ((start-index 1)
             (column-header-map #s(hash-table test equal data ("No" ((:align . 'left) (:title . "No"))))))
         (mapcar
          (lambda (item)
            (puthash
             (format "%s" item)
             `((:align . ,'left) (:title . ,item))
             column-header-map))
          (cond ((string= org-type "nonScratchOrgs") sfmm:non-scratch-org-header-display)
                ((string= org-type "scratchOrgs")) sfmm:scratch-org-header-display))

         (if (= (gethash "status" json-instance) 0)
             (let* ((non-scratch-orgs
                     (sfdx--helper:get-data-hashtable
                      (concat "result." org-type) json-instance))
                    (data
                     (mapcar
                      (lambda (element)
                        (let ((row
                               (sfdx--helper:convert-hashtable-data-to-list
                                :hashtable-data element
                                :columns
                                (cond ((string= org-type "nonScratchOrgs") sfmm:non-scratch-org-header-display)
                                      ((string= org-type "scratchOrgs")) sfmm:scratch-org-header-display)
                                :post-process
                                (lambda (key value)
                                  (let ((column-config
                                         `((:align . ,'left) (:title . ,key))))

                                    (when (length> value 50)
                                        (add-to-list 'column-config `(:max-width . ,'50))

                                        (puthash (format "%s" key) column-config column-header-map)))))))

                          (progn
                            (add-to-list 'row start-index
                                         nil (lambda (el1 el2) nil))
                            (setq start-index
                                  (1+ start-index))
                            row)))
                      non-scratch-orgs)))

               (sfdx--build:make-table-component
                :column-header (hash-table-values column-header-map)
                :data data
                :buffer buffer))))))))

(defun sfmm:create-apex-class ()
  "Generate apex class"
  (interactive)
  (let* ((class-name (read-string "class name: "))
         (class-extend (read-string "class parent: "))
         (class-implements (read-string "class implements: "))
         (command (sfdx--helper:generate-command
                   (list sfmm:apex-command-alias "generate" "class" "--name" class-name "--output-dir" (projectile-expand-root sfmm:default-apex-class-path "--json"))))

    (progn
      (unless (string= class-extend "")
        (setq class-expand (concat class-expand "extends" " " class-extend " ")))
      (unless (string= class-implements "")
        (setq class-expand (concat class-expand "implements" " " class-implements " "))))

    (sfdx--helper:make-sync-process
     :command command
     :handle-success-lambda
     `(lambda (process json-instance buffer)

        (shell-command
         ,(format "sed -ir 's/(public|private|global)\\s?(abstract|virtual)?\\s?(without sharing|with sharing)?\\s?(class|interface)\\s?(\\S*)/\0 %s/g' %s"
                 (concat sfmm:default-apex-class-path "/" class-name ".cls")))))))

*** EXCUTE SOQL BLOCK
This function execute [[https://developer.salesforce.com/docs/atlas.en-us.sfdx_cli_reference.meta/sfdx_cli_reference/cli_reference_force_data.htm#cli_reference_force_data_soql_query][command]] from library [[https://developer.salesforce.com/docs/atlas.en-us.sfdx_setup.meta/sfdx_setup/sfdx_setup_intro.htm][Salesforce CLI]]
- Reference: sfdx-fetch-record-through-soql
- Params:
+ params-command: A list of params is used by Salesforce SOQL command, supporting the following type:
    + user: the user is used to executing command
    + api-version: version API is using
    + query: SOQL string want to execute
    + file: file contains chain SOQL want to execute
    + format: format for result return from API

**** Body
#+begin_src emacs-lisp :tangle yes
(cl-defun sfdx--helper:execute-soql (&key query (type 'string) (options '("--json")))
  "Excute command fetch records from Salesforce through API"
  (progn
    (cond ((string= type 'string)
           (setq options (append options (list "--query" query))))
          ((string= type 'file)
           (setq options (append options (list "--file" query)))))
    (let* ((execute-soql-code
            (append (sfdx--helper:generate-command (list sfmm:data-command-alias "query"))
                    options)))

      (sfdx--helper:make-sync-process
       :command execute-soql-code
       :handle-success-lambda
       (lambda (process json-instance buffer)
         (let* ((records-list (sfdx--helper:get-data-hashtable "result.records" json-instance))
                (start-index 1)
                (column-header-map #s(hash-table test equal data ("No" ((:align . 'left) (:title . "No")))))
                (data
                 (mapcar
                  (lambda (element)

                    (let ((row `(,start-index)))
                      (maphash
                       (lambda (key value)
                         (let ((column-config `((:align . ,'left) (:title . ,key))))
                           (unless (hash-table-p value)
                             (when (eq value ':null)
                               (setq value " "))
                             (when (eq value ':false)
                               (setq value "False"))
                             (when (eq value 't)
                               (setq value "True"))

                             (add-to-list 'row
                                          (replace-regexp-in-string "\\([\n]\\)" "" value) 1
                                          '(lambda (element1 element2) nil))

                             (when (length> value 50)
                               (add-to-list 'column-config `(:max-width . ,'50)))

                             (puthash (format "%s" key) column-config column-header-map))))
                       element)
                      (setq start-index (1+ start-index))
                      row))
                  records-list)))
           (sfdx--build:make-table-component
            :column-header (hash-table-values column-header-map)
            :data data
            :buffer buffer)))))))

#+begin_src emacs-lisp :tangle yes
(cl-defun sfdx--helper:make-sync-process (&key command handle-success-lambda (handle-error-lambda))

(cl-defun sfdx--helper:make-async-process
    (&key command handle-success-lambda handle-error-lambda)
  "Make async process"
  (let ((buffer-command-history "salesforce-command-history")
        (buffer-stdout (generate-new-buffer "salesforce-process"))
        (process-identity "salesforce-process")
        (buffer-view-post-process (generate-new-buffer "salesforce outcome")))

      (make-process
       :name process-identity
       :buffer buffer-command-history

       `(lambda (process output)
          (with-current-buffer ,buffer-stdout
             (insert
              (concat (buffer-string) output))))
       :sentinel
       `(lambda (process event)
          (cond ((string= event "finished\n")

                 (with-current-buffer ,buffer-stdout

                               (funcall ,handle-error-lambda process json-instance ,buffer-view-post-process)

                             (alert (format "context: %s \nmessage: %s "
                                            (gethash "context" json-instance)
                                            (gethash "message" json-instance)
                                            :title "Salesforce Alert"))))))))
                ((string= event "deleted\n")
                 (kill-buffer (process-buffer process)))))))))

(cl-defun sfdx--helper:convert-hashtable-data-to-list
    (&key hashtable-data columns (post-process nil))
  "Convert hashtable data to list"
 (let ((data '()))

   (mapcar
      `(lambda (key)
         (when (member key columns)
           (let ((value (gethash key hashtable-data)))

             (when (eq value ':null)
               (setq value " "))
             (when (eq value ':false)
               (setq value "False"))
             (when (eq value 't)
               (setq value "True"))
             (when (eq value 'nil)
               (setq value " "))

             (add-to-list 'data value 1
                          '(lambda (element1 element2) nil))

             (when ,post-process
               (funcall ,post-process key value)))))
      columns)
   data))

(defun sfdx--helper:recursive-list (list-data lambda-function)
  (let* ((new-list (cdr list-data))
         (first-item (car list-data))
         (remap-list '()))

    (when (length> new-list 0)
        (setq remap-list
              (append remap-list
                      (sfdx--helper:recursive-list new-list lambda-function))))

    (add-to-list 'remap-list (funcall lambda-function first-item))))

(defun sfdx--helper:get-data-hashtable (path table)
  "Get all data follow the path in hash table"
  (let* ((path-splited (split-string path "\\."))
         (key (car path-splited))
         (key-remain (cdr path-splited)))

    (if key-remain
      (sfdx--helper:get-data-hashtable
       (string-join key-remain ".")
       (gethash key table))

     (gethash key table))))

(defun sfdx--build:make-header-table (column)
  "Build header ctable"
  (let* ((header-config column)
         (title (if (assq ':title column)
                    (cdr (assq ':title column))
                  (car column)))
         (min-width
          (cdr
           (assq ':min-width header-config)))
         (max-width
          (cdr
           (assq ':max-width header-config)))
         (align
          (cdr
           (assq ':align header-config))))

    (make-ctbl:cmodel
     :title title
     :sorter 'ctbl:sort-number-lessp
     :min-width min-width
     :max-width max-width
     :align align)))
