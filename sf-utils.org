** PUSH SOURCE SALESFORCE TO ORG
#+begin_src emacs-lisp :tangle yes
(defun sfdx-push ()
   "Push source salesforce to org"
   (interactive)
   (let
    ((push-command (message "sfdx force:source:deploy -p %s" buffer-file-name)))
    (async-shell-command push-command buffer)))
#+end_src

** AUTHORIZE SALESFORCE ORG
#+begin_src emacs-lisp :tangle yes
(defun sfdx-authorize-web ()
   "Authorize to salesforce org"
   (interactive)
   (let
    ((url-list '("https://test.salesforce.com" "https://salesforce.com"))
     (alias (read-string "alias: "))
     (url (read-string "url: " "https://test.salesforce.com")))
    (setq authorize-command
      (message "sfdx auth:web:login --setalias %s --instanceurl %s" alias url))
    (async-shell-command authorize-command buffer)))
#+end_src
