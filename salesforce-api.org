*** EXCUTE SOQL BLOCK
This function execute [[https://developer.salesforce.com/docs/atlas.en-us.sfdx_cli_reference.meta/sfdx_cli_reference/cli_reference_force_data.htm#cli_reference_force_data_soql_query][command]] from library [[https://developer.salesforce.com/docs/atlas.en-us.sfdx_setup.meta/sfdx_setup/sfdx_setup_intro.htm][Salesforce CLI]]
+ Reference: sfdx-fetch-record-through-soql
+ Params:
  * params-command: A list of params is used by Salesforce SOQL command, supporting the following type:
    + user: the user is used to executing command
    + api-version: version API is using
    + query: SOQL string want to execute
    + file: file contains chain SOQL want to execute
    + format: format for result return from API

#+begin_src emacs-lisp :tangle yes
(defun sfdx--execute-soql (params-command)
  "Excute command fetch records from Salesforce through API"
  (let* ((param-support
            '((:user . "-u")
              (:api-version . "--apiversion")
              (:query . "-q")
              (:file . "-f")
              (:format . "-r")))
         (params
             (sfdx--recursive-list
               params-command
               `(lambda (item)
                  (let* ((key (car item)))
                    (when (assq key param-support)
                      (format "%s \'%s\'"
                       (cdr (assq key param-support))
                       (cdr item)))))))
         (execute-soql-code
           (message "sfdx force:data:soql:query %s" (string-join params " "))))

        (sfdx--execute-process
         execute-soql-code
         `(lambda (process output buffer)
                (if (json-available-p)
                    (with-current-buffer (find-file (buffer-name buffer))
                      (set-buffer-major-mode buffer)
                      (save-excursion
                        (let* ((json-instance (json-parse-string output)))
                          (when (= (gethash "status" json-instance) 0)
                                (let* ((result-instance (gethash "result" json-instance))
                                       (records-list (gethash "records" result-instance))
                                       (columns-name '())
                                       (j2t
                                         (mapconcat
                                          (lambda (element)
                                             (let ((columns-data '()))

                                               (maphash
                                                (lambda (key value)
                                                  (unless (hash-table-p value)
                                                    (when (keywordp value)
                                                      (when (eq value ':null)
                                                        (setq value " "))
                                                      (when (eq value ':false)
                                                        (setq value "False"))
                                                      (when (eq value ':t)
                                                        (setq value "True")))

                                                    (add-to-list 'columns-data
                                                                 (replace-regexp-in-string "\\([\n]\\)" "" value)
                                                                 1)
                                                    (add-to-list 'columns-name key 1)))
                                                element)

                                              (concat "|" (string-join columns-data "|") "|")))
                                          records-list
                                          "\n")))

                                  (insert
                                   (format "|%s|\n|----\n%s" (string-join columns-name "|") j2t))
                                  (org-table-align)
                                  (read-only-mode)))))))))))

(defun sfdx--execute-process (command lambda-function)
  (let* ((buffer-stdout (format "*%s*" "salesforce-api"))
         (process-identity "salesforce-api")
         (buffer-view-post-process (generate-new-buffer "Salesforce Overview"))
         (process-output "")
         (process
          (start-process-shell-command process-identity buffer-stdout command)))

        (set-process-filter
         process
         `(lambda (process output)
            (setq process-output (concat process-output output))))
        (when (accept-process-output process))
        ; clear old content
        (with-current-buffer buffer-stdout
          (setf (buffer-string) ""))
        (funcall lambda-function process process-output buffer-view-post-process)))

(defun sfdx--recursive-list (list-data lambda-function)
  (let* ((new-list (cdr list-data))
         (first-item (car list-data))
         (remap-list '()))

    (when (length> new-list 0)
        (setq remap-list
              (append remap-list
                      (sfdx--recursive-list new-list lambda-function))))

    (add-to-list 'remap-list (funcall lambda-function first-item))))
