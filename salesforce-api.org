(defun sfdx-display-org (org-type)
  "Display all current connect org"
  (interactive)
  (let ((command '("sfdx" "force:org:list" "--json")))

    (sfdx--helper:execute-process
     command
     `(lambda (process output buffer)
        (if (json-available-p)
            (let ((json-instance (json-parse-string output))
                  (start-index 1)
                  (column-header-map #s(hash-table test equal data ("No" ((:align . 'left) (:title . "No"))))))
              (mapcar
               `(lambda (item)
                  (puthash
                   (format "%s" item)
                   `((:align . ,'left) (:title . ,item))
                   column-header-map))
               sfdx:org-header-display)

              (if (= (gethash "status" json-instance) 0)
                  (let* ((non-scratch-orgs
                          (sfdx--helper:get-data-hashtable
                           (concat "result." org-type) json-instance))
                         (data
                          (mapcar
                           (lambda (element)
                             (let ((row
                                    (sfdx--helper:convert-hashtable-data-to-list
                                     :hashtable-data element
                                     :columns
                                     sfdx:org-header-display
                                     :post-process
                                     (lambda (key value)
                                       (let ((column-config
                                              `((:align . ,'left) (:title . ,key))))

                                         (when (length> value 50)
                                             (add-to-list 'column-config `(:max-width . ,'50)))

                                         (puthash (format "%s" key) column-config column-header-map))))))

                               (progn
                                 (add-to-list 'row start-index
                                             nil (lambda (el1 el2) nil))
                                 (setq start-index
                                       (1+ start-index))
                                 row)))
                           non-scratch-orgs)))

                    (sfdx--build:make-table-component
                               :column-header (hash-table-values column-header-map)
                               :data data
                               :buffer buffer)))))))))

*** EXCUTE SOQL BLOCK
This function execute [[https://developer.salesforce.com/docs/atlas.en-us.sfdx_cli_reference.meta/sfdx_cli_reference/cli_reference_force_data.htm#cli_reference_force_data_soql_query][command]] from library [[https://developer.salesforce.com/docs/atlas.en-us.sfdx_setup.meta/sfdx_setup/sfdx_setup_intro.htm][Salesforce CLI]]
**** Reference: sfdx-fetch-record-through-soql
**** Params:
+ params-command: A list of params is used by Salesforce SOQL command, supporting the following type:
    + user: the user is used to executing command
    + api-version: version API is using
    + query: SOQL string want to execute
    + file: file contains chain SOQL want to execute
    + format: format for result return from API

**** Body
#+begin_src emacs-lisp :tangle yes
(defun sfdx--helper:execute-soql (params-command)
  "Excute command fetch records from Salesforce through API"
  (let* ((param-support
            '((:user . "-u")
              (:api-version . "--apiversion")
              (:query . "-q")
              (:file . "-f")
              (:format . "-r")))
         (execute-soql-code
          (append '("sfdx" "force:data:soql:query"))))

    (sfdx--helper:recursive-list
      params-command
      (lambda (item)
        (let* ((key (car item)))
          (when (assq key param-support)
            (setq
             execute-soql-code
             (append
              execute-soql-code
              (list
               (cdr (assq key param-support))
               (cdr item))))))))

    (sfdx--helper:execute-process
     execute-soql-code
     `(lambda (process output buffer)
            (if (json-available-p)
                (let* ((json-instance (json-parse-string output)))
                  (when (= (gethash "status" json-instance) 0)
                        (let* ((records-list (sfdx--helper:get-data-hashtable "result.records" json-instance))
                               (start-index 1)
                               (column-header-map #s(hash-table test equal data ("No" ((:align . 'left) (:title . "No")))))
                               (data
                                 (mapcar
                                  (lambda (element)

                                     (let ((row `(,start-index)))
                                       (maphash
                                        (lambda (key value)
                                          (let ((column-config `((:align . ,'left) (:title . ,key))))
                                            (unless (hash-table-p value)
                                               (when (eq value ':null)
                                                 (setq value " "))
                                               (when (eq value ':false)
                                                 (setq value "False"))
                                               (when (eq value 't)
                                                 (setq value "True"))

                                               (add-to-list 'row
                                                            (replace-regexp-in-string "\\([\n]\\)" "" value) 1
                                                            '(lambda (element1 element2) nil))

                                               (when (length> value 50)
                                                 (add-to-list 'column-config `(:max-width . ,'50)))

                                               (puthash (format "%s" key) column-config column-header-map))))
                                        element)
                                       (setq start-index (1+ start-index))
                                       row))
                                  records-list)))
                          (sfdx--build:make-table-component
                           :column-header (hash-table-values column-header-map)
                           :data data
                           :buffer buffer)))))))))

(defun sfdx--helper:execute-process (command lambda-function)
  (let* ((buffer-stdout (format "*%s*" "salesforce-api"))
         (process-identity "salesforce-api")
         (buffer-view-post-process (generate-new-buffer "Salesforce Overview"))
         (process-output "")
         (process
          (make-process
           :name process-identity
           :buffer buffer-stdout
           :command command
           :connection-type 'pipe
           :filter
           (lambda (process output)
             (setq process-output (concat process-output output))))))
    (when (accept-process-output process))
        ; clear old content
    (with-current-buffer buffer-stdout
      (setf (buffer-string) ""))
    (funcall lambda-function process process-output buffer-view-post-process)))

*** CONVERT HASHTABLE TO LIST DATA
#+begin_src emacs-lisp :tangle yes
(cl-defun sfdx--helper:convert-hashtable-data-to-list
    (&key hashtable-data columns &option post-process)
  "Convert hashtable data to list"
 (let ((data '()))

   (mapcar
      `(lambda (key)
         (when (member key columns)
           (let ((value (gethash key hashtable-data)))

             (when (eq value ':null)
               (setq value " "))
             (when (eq value ':false)
               (setq value "False"))
             (when (eq value 't)
               (setq value "True"))
             (when (eq value 'nil)
               (setq value " "))

             (add-to-list 'data value 1
                          '(lambda (element1 element2) nil))

             (funcall ,post-process key value))))
      columns)
   data))
#+end_src

#+begin_src emacs-lisp :tangle yes
(defun sfdx--helper:recursive-list (list-data lambda-function)
  (let* ((new-list (cdr list-data))
         (first-item (car list-data))
         (remap-list '()))

    (when (length> new-list 0)
        (setq remap-list
              (append remap-list
                      (sfdx--helper:recursive-list new-list lambda-function))))

    (add-to-list 'remap-list (funcall lambda-function first-item))))
#+end_src

(defun sfdx--helper:get-data-hashtable (path table)
  "Get all data follow the path in hash table"
  (let* ((path-splited (split-string path "\\."))
         (key (car path-splited))
         (key-remain (cdr path-splited)))

    (if key-remain
      (sfdx--helper:get-data-hashtable
       (string-join key-remain "\\.")
       (gethash key table))

     (gethash key table))))

(cl-defun sfdx--build:make-table-component (&key column-header data buffer)
  "Use ctable to build table data"
  (let* ((column-model (mapcar 'sfdx--build:make-header-table column-header))
         (async-model
          (ctbl:async-model-wrapper data))
         (model
          (make-ctbl:model
           :column-model column-model :data async-model)))

    (with-current-buffer (pop-to-buffer buffer)
      (ctbl:create-table-component-region
           :model model)
      (ctbl:table-mode)
      (beginning-of-buffer)
      (read-only-mode))))

#+begin_src emacs-lisp :tangle yes
(defun sfdx--build:make-header-table (column)
  "Build header ctable"
  (let* ((header-config column)
         (title (if (assq ':title column)
                    (cdr (assq ':title column))
                  (car column)))
         (min-width
          (cdr
           (assq ':min-width header-config)))
         (max-width
          (cdr
           (assq ':max-width header-config)))
         (align
          (cdr
           (assq ':align header-config))))

    (make-ctbl:cmodel
     :title title
     :sorter 'cbtbl:sort-number-lessp
     :min-width min-width
     :max-width max-width
     :align align)))
