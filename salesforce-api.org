(defun sfdx-push-file ()
   "Push source salesforce to org"
   (interactive)
   (let
    ((push-command (message "sfdx force:source:deploy -p %s" buffer-file-name)))
    (execute-command push-command (format "Deploy %s success" (buffer-file-name)))))

(defun sfdx-authorize-web ()
   "Authorize to salesforce org"
   (interactive)
   (let
    ((url-list '("https://test.salesforce.com" "https://salesforce.com"))
     (alias (read-string "alias: "))
     (url (read-string "url: " "https://test.salesforce.com")))
    (setq authorize-command
      (message "sfdx auth:web:login --setalias %s --instanceurl %s" alias url))
    (execute-command authorize-command "Authorize success")))

(defun sfdx-execute-apex (content)
  "Execute code"
  (interactive)
  (let
   ((code
     (unless
      content
      (point-min)
      content))
    (temp-file (make-temp-file "temp_code")))

   (write-region (point-min) (point-max) temp-file)
   (setq execute-apex-code-command
     (message "sfdx force:apex:execute -f %s" temp-file))

   (async-shell-command execute-apex-code-command buffer)))

*** EXECUTE SALESFORCE COMMAND
#+begin_src emacs-lisp :tangle yes
(defun execute-command (command message-success)
  (let* ((buffer-name (format "*%s*" "salesforce-api"))
         (process-name "salesforce-api")
         (process
          (start-process-shell-command process-name buffer-name command)))

        (set-process-sentinel
         process
         `(lambda (process event)
            (when (string= "finished\n" event)
              (kill-buffer ,buffer-name)
              (alert ,message-success :title "Salesforce Alert"))))))
#+end_src
